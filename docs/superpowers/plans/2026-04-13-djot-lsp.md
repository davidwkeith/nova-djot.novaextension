# Djot LSP Server Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a Go-based LSP server to the Nova Djot extension, providing completions, diagnostics, document symbols, hover, and go-to-definition.

**Architecture:** Go binary using `tliron/glsp` for LSP scaffolding and `sivukhin/godjot/v2` for Djot parsing. Nova's `LanguageClient` spawns the binary via stdio. Document is re-parsed on every change; AST is walked to build indexes for references, footnotes, and headings.

**Tech Stack:** Go 1.23+, tliron/glsp (LSP framework), sivukhin/godjot/v2 (Djot parser), Nova LanguageClient API

**Spec:** `docs/superpowers/specs/2026-04-13-djot-lsp-design.md`

```go
package server

import (
	"sync"

	"github.com/tliron/glsp"
	protocol "github.com/tliron/glsp/protocol_3_16"
)

var documents sync.Map // uri string → *Document

func NewHandler() *protocol.Handler {
	handler := &protocol.Handler{
		Initialize:             handleInitialize,
		Initialized:            handleInitialized,
		Shutdown:               handleShutdown,
		TextDocumentDidOpen:    handleDidOpen,
		TextDocumentDidChange:  handleDidChange,
		TextDocumentDidClose:   handleDidClose,
	}
	return handler
}

func handleInitialize(ctx *glsp.Context, params *protocol.InitializeParams) (any, error) {
	return protocol.InitializeResult{
		Capabilities: protocol.ServerCapabilities{
			TextDocumentSync: &protocol.TextDocumentSyncOptions{
				OpenClose: boolPtr(true),
				Change:    protocol.TextDocumentSyncKindFull,
			},
			CompletionProvider: &protocol.CompletionOptions{
				TriggerCharacters: []string{"[", "#"},
			},
			HoverProvider:          true,
			DefinitionProvider:     true,
			DocumentSymbolProvider: true,
		},
	}, nil
}

func handleInitialized(ctx *glsp.Context, params *protocol.InitializedParams) error {
	return nil
}

func handleShutdown(ctx *glsp.Context) error {
	return nil
}

func handleDidOpen(ctx *glsp.Context, params *protocol.DidOpenTextDocumentParams) error {
	doc := NewDocument(params.TextDocument.URI, params.TextDocument.Text)
	documents.Store(params.TextDocument.URI, doc)
	return publishDiagnostics(ctx, doc)
}

func handleDidChange(ctx *glsp.Context, params *protocol.DidChangeTextDocumentParams) error {
	if len(params.ContentChanges) == 0 {
		return nil
	}
	// Full sync — last content change has the full text
	text := params.ContentChanges[len(params.ContentChanges)-1].(protocol.TextDocumentContentChangeEventWholeDocument).Text
	doc := NewDocument(params.TextDocument.URI, text)
	documents.Store(params.TextDocument.URI, doc)
	return publishDiagnostics(ctx, doc)
}

func handleDidClose(ctx *glsp.Context, params *protocol.DidCloseTextDocumentParams) error {
	documents.Delete(params.TextDocument.URI)
	// Clear diagnostics
	return ctx.Notify(protocol.ServerTextDocumentPublishDiagnostics, protocol.PublishDiagnosticsParams{
		URI:         params.TextDocument.URI,
		Diagnostics: []protocol.Diagnostic{},
	})
}

func getDocument(uri string) *Document {
	val, ok := documents.Load(uri)
	if !ok {
		return nil
	}
	return val.(*Document)
}

func boolPtr(b bool) *bool {
	return &b
}
```

```go
package server

import (
	"sort"
	"strings"

	"github.com/sivukhin/godjot/v2/djot_parser"
	protocol "github.com/tliron/glsp/protocol_3_16"
)

type HeadingInfo struct {
	Label    string
	Level    int
	ID       string
	Position protocol.Position
	Range    protocol.Range
}

type DefinitionInfo struct {
	Label       string
	Destination string
	Position    protocol.Position
	Range       protocol.Range
	Text        string
}

type ReferenceInfo struct {
	Label    string
	Position protocol.Position
	Range    protocol.Range
}

type Document struct {
	URI         string
	Source      string
	LineOffsets []int
	AST         []djot_parser.TreeNode[djot_parser.DjotNode]

	Headings      []HeadingInfo
	FootnoteDefs  map[string]DefinitionInfo
	FootnoteRefs  []ReferenceInfo
	ReferenceDefs map[string]DefinitionInfo
	ReferenceRefs []ReferenceInfo
}

func NewDocument(uri string, source string) *Document {
	doc := &Document{
		URI:           uri,
		Source:        source,
		LineOffsets:   buildLineOffsets(source),
		FootnoteDefs:  make(map[string]DefinitionInfo),
		ReferenceDefs: make(map[string]DefinitionInfo),
	}

	doc.AST = djot_parser.BuildDjotAst([]byte(source))
	doc.buildIndexes()
	return doc
}

func buildLineOffsets(source string) []int {
	offsets := []int{0}
	for i, ch := range source {
		if ch == '\n' {
			offsets = append(offsets, i+1)
		}
	}
	return offsets
}

func (d *Document) OffsetToPosition(offset int) protocol.Position {
	if offset < 0 {
		offset = 0
	}
	if offset > len(d.Source) {
		offset = len(d.Source)
	}
	line := sort.SearchInts(d.LineOffsets, offset+1) - 1
	if line < 0 {
		line = 0
	}
	character := offset - d.LineOffsets[line]
	return protocol.Position{
		Line:      protocol.UInteger(line),
		Character: protocol.UInteger(character),
	}
}

func (d *Document) PositionToOffset(pos protocol.Position) int {
	line := int(pos.Line)
	if line >= len(d.LineOffsets) {
		return len(d.Source)
	}
	return d.LineOffsets[line] + int(pos.Character)
}

func (d *Document) buildIndexes() {
	for _, node := range d.AST {
		d.walkNode(node)
	}
}

func (d *Document) walkNode(node djot_parser.TreeNode[djot_parser.DjotNode]) {
	switch node.Type {
	case djot_parser.HeadingNode:
		level := 1
		if lvl := node.Attributes.Get(djot_parser.HeadingLevelKey); lvl != "" {
			for _, ch := range lvl {
				if ch == '#' {
					level++
				}
			}
			level = len(lvl)
		}
		label := strings.TrimSpace(string(node.FullText()))
		id := node.Attributes.Get("id")
		d.Headings = append(d.Headings, HeadingInfo{
			Label: label,
			Level: level,
			ID:    id,
		})

	case djot_parser.FootnoteDefNode:
		label := node.Attributes.Get("id")
		if label == "" {
			label = node.Attributes.Get(djot_parser.ReferenceKey)
		}
		if label != "" {
			// Strip "fn-" prefix if present
			cleanLabel := strings.TrimPrefix(label, "fn-")
			d.FootnoteDefs[cleanLabel] = DefinitionInfo{
				Label: cleanLabel,
				Text:  strings.TrimSpace(string(node.FullText())),
			}
		}

	case djot_parser.ReferenceDefNode:
		label := node.Attributes.Get(djot_parser.ReferenceKey)
		dest := node.Attributes.Get(djot_parser.LinkDestinationKey)
		if label != "" {
			d.ReferenceDefs[label] = DefinitionInfo{
				Label:       label,
				Destination: dest,
			}
		}

	case djot_parser.LinkNode:
		ref := node.Attributes.Get(djot_parser.ReferenceKey)
		if ref != "" {
			if strings.HasPrefix(ref, "fn-") {
				d.FootnoteRefs = append(d.FootnoteRefs, ReferenceInfo{
					Label: strings.TrimPrefix(ref, "fn-"),
				})
			} else {
				d.ReferenceRefs = append(d.ReferenceRefs, ReferenceInfo{
					Label: ref,
				})
			}
		}
	}

	for _, child := range node.Children {
		d.walkNode(child)
	}
}
```

**Note:** The exact godjot attribute keys (`HeadingLevelKey`, `ReferenceKey`, `LinkDestinationKey`) need to be verified against the godjot source. The implementer should read `djot_parser/djot_ast.go` to find the correct constant names. If they differ, adjust accordingly.

```go
package server

import (
	"fmt"

	"github.com/tliron/glsp"
	protocol "github.com/tliron/glsp/protocol_3_16"
)

func publishDiagnostics(ctx *glsp.Context, doc *Document) error {
	diags := computeDiagnostics(doc)
	return ctx.Notify(protocol.ServerTextDocumentPublishDiagnostics, protocol.PublishDiagnosticsParams{
		URI:         doc.URI,
		Diagnostics: diags,
	})
}

func computeDiagnostics(doc *Document) []protocol.Diagnostic {
	var diags []protocol.Diagnostic

	// Undefined footnote references
	for _, ref := range doc.FootnoteRefs {
		if _, ok := doc.FootnoteDefs[ref.Label]; !ok {
			diags = append(diags, protocol.Diagnostic{
				Range:    ref.Range,
				Severity: &warningDiagSeverity,
				Source:   strPtr("djot-lsp"),
				Message:  fmt.Sprintf("Undefined footnote reference: %s", ref.Label),
			})
		}
	}

	// Undefined link references
	for _, ref := range doc.ReferenceRefs {
		if _, ok := doc.ReferenceDefs[ref.Label]; !ok {
			// Also check if it's a heading auto-reference
			found := false
			for _, h := range doc.Headings {
				if h.ID == ref.Label {
					found = true
					break
				}
			}
			if !found {
				diags = append(diags, protocol.Diagnostic{
					Range:    ref.Range,
					Severity: &warningDiagSeverity,
					Source:   strPtr("djot-lsp"),
					Message:  fmt.Sprintf("Undefined link reference: %s", ref.Label),
				})
			}
		}
	}

	// Duplicate footnote definitions
	// (Document.FootnoteDefs is a map so duplicates overwrite — need to detect during walk)
	diags = append(diags, doc.DuplicateDiagnostics...)

	return diags
}

var warningDiagSeverity = protocol.DiagnosticSeverityWarning

func strPtr(s string) *string {
	return &s
}
```

**Note:** Duplicate detection requires tracking during the AST walk. Add a `DuplicateDiagnostics []protocol.Diagnostic` field to `Document` and populate it in `buildIndexes()` when a label already exists in the map. The implementer should update `document.go` accordingly:

In `document.go`, update the `Document` struct to include:

And in the `FootnoteDefNode` and `ReferenceDefNode` cases of `walkNode`, check if the label already exists before storing:

```go
package server

import (
	"fmt"

	"github.com/tliron/glsp"
	protocol "github.com/tliron/glsp/protocol_3_16"
)

func init() {
	// Will be registered in server.go handler
}

func handleDocumentSymbol(ctx *glsp.Context, params *protocol.DocumentSymbolParams) (any, error) {
	doc := getDocument(params.TextDocument.URI)
	if doc == nil {
		return nil, nil
	}
	symbols := computeDocumentSymbols(doc)
	return symbols, nil
}

func computeDocumentSymbols(doc *Document) []protocol.DocumentSymbol {
	if len(doc.Headings) == 0 {
		return nil
	}

	var result []protocol.DocumentSymbol
	var stack []*protocol.DocumentSymbol

	for _, h := range doc.Headings {
		sym := protocol.DocumentSymbol{
			Name:           h.Label,
			Detail:         fmt.Sprintf("H%d", h.Level),
			Kind:           protocol.SymbolKindString,
			Range:          h.Range,
			SelectionRange: h.Range,
		}

		// Pop stack until we find a parent with lower level
		for len(stack) > 0 {
			parentIdx := len(stack) - 1
			parentLevel := 0
			fmt.Sscanf(stack[parentIdx].Detail, "H%d", &parentLevel)
			if parentLevel < h.Level {
				break
			}
			stack = stack[:parentIdx]
		}

		if len(stack) == 0 {
			result = append(result, sym)
			stack = append(stack, &result[len(result)-1])
		} else {
			parent := stack[len(stack)-1]
			parent.Children = append(parent.Children, sym)
			stack = append(stack, &parent.Children[len(parent.Children)-1])
		}
	}

	return result
}
```

```go
package server

import (
	"fmt"
	"strings"

	"github.com/tliron/glsp"
	protocol "github.com/tliron/glsp/protocol_3_16"
)

func handleCompletion(ctx *glsp.Context, params *protocol.CompletionParams) (any, error) {
	doc := getDocument(params.TextDocument.URI)
	if doc == nil {
		return nil, nil
	}
	items := computeCompletions(doc, params.Position)
	return items, nil
}

func computeCompletions(doc *Document, pos protocol.Position) []protocol.CompletionItem {
	offset := doc.PositionToOffset(pos)
	if offset < 2 {
		return nil
	}

	// Look at the text before the cursor to determine context
	lineStart := doc.LineOffsets[int(pos.Line)]
	textBeforeCursor := doc.Source[lineStart:offset]

	// Footnote completion: triggered after "[^"
	if strings.HasSuffix(textBeforeCursor, "[^") {
		return footnoteCompletions(doc)
	}

	// Reference link completion: triggered after "]["
	if strings.HasSuffix(textBeforeCursor, "][") {
		return referenceCompletions(doc)
	}

	// Heading ID completion: triggered after "{#" inside attributes
	if strings.HasSuffix(textBeforeCursor, "{#") || strings.HasSuffix(textBeforeCursor, " #") {
		// Check if we're inside an attribute block
		if strings.Contains(textBeforeCursor, "{") {
			return headingIDCompletions(doc)
		}
	}

	return nil
}

func footnoteCompletions(doc *Document) []protocol.CompletionItem {
	var items []protocol.CompletionItem
	for label, def := range doc.FootnoteDefs {
		detail := def.Text
		if len(detail) > 80 {
			detail = detail[:80] + "..."
		}
		items = append(items, protocol.CompletionItem{
			Label:  label,
			Kind:   completionKindPtr(protocol.CompletionItemKindReference),
			Detail: &detail,
			InsertText: strPtr(fmt.Sprintf("%s]", label)),
		})
	}
	return items
}

func referenceCompletions(doc *Document) []protocol.CompletionItem {
	var items []protocol.CompletionItem
	for label, def := range doc.ReferenceDefs {
		detail := def.Destination
		items = append(items, protocol.CompletionItem{
			Label:  label,
			Kind:   completionKindPtr(protocol.CompletionItemKindReference),
			Detail: &detail,
			InsertText: strPtr(fmt.Sprintf("%s]", label)),
		})
	}
	return items
}

func headingIDCompletions(doc *Document) []protocol.CompletionItem {
	var items []protocol.CompletionItem
	for _, h := range doc.Headings {
		if h.ID != "" {
			detail := fmt.Sprintf("H%d: %s", h.Level, h.Label)
			items = append(items, protocol.CompletionItem{
				Label:  h.ID,
				Kind:   completionKindPtr(protocol.CompletionItemKindValue),
				Detail: &detail,
			})
		}
	}
	return items
}

func completionKindPtr(k protocol.CompletionItemKind) *protocol.CompletionItemKind {
	return &k
}
```

```go
package server

import (
	"fmt"
	"strings"

	"github.com/tliron/glsp"
	protocol "github.com/tliron/glsp/protocol_3_16"
)

func handleHover(ctx *glsp.Context, params *protocol.HoverParams) (*protocol.Hover, error) {
	doc := getDocument(params.TextDocument.URI)
	if doc == nil {
		return nil, nil
	}
	return computeHover(doc, params.Position), nil
}

func computeHover(doc *Document, pos protocol.Position) *protocol.Hover {
	offset := doc.PositionToOffset(pos)
	
	// Find what's at the cursor position by scanning the source
	// Look for footnote references: [^label]
	if ref := findFootnoteRefAt(doc, offset); ref != "" {
		if def, ok := doc.FootnoteDefs[ref]; ok {
			return &protocol.Hover{
				Contents: protocol.MarkupContent{
					Kind:  protocol.MarkupKindMarkdown,
					Value: fmt.Sprintf("**Footnote [^%s]:**\n\n%s", ref, def.Text),
				},
			}
		}
		return &protocol.Hover{
			Contents: protocol.MarkupContent{
				Kind:  protocol.MarkupKindMarkdown,
				Value: fmt.Sprintf("**Footnote [^%s]:** _undefined_", ref),
			},
		}
	}

	// Look for reference links: [text][ref]
	if ref := findLinkRefAt(doc, offset); ref != "" {
		if def, ok := doc.ReferenceDefs[ref]; ok {
			return &protocol.Hover{
				Contents: protocol.MarkupContent{
					Kind:  protocol.MarkupKindMarkdown,
					Value: fmt.Sprintf("**[%s]:** %s", ref, def.Destination),
				},
			}
		}
	}

	// Look for inline links: [text](url)
	if url := findInlineLinkAt(doc, offset); url != "" {
		return &protocol.Hover{
			Contents: protocol.MarkupContent{
				Kind:  protocol.MarkupKindMarkdown,
				Value: url,
			},
		}
	}

	return nil
}

func findFootnoteRefAt(doc *Document, offset int) string {
	// Search backward for "[^" and forward for "]"
	src := doc.Source
	start := offset
	for start > 0 && src[start] != '[' {
		start--
	}
	if start+1 < len(src) && src[start] == '[' && src[start+1] == '^' {
		end := strings.IndexByte(src[start:], ']')
		if end > 0 && offset <= start+end {
			return src[start+2 : start+end]
		}
	}
	return ""
}

func findLinkRefAt(doc *Document, offset int) string {
	// Search for "][label]" pattern around offset
	src := doc.Source
	// Find the "]" + "[" boundary
	pos := offset
	for pos > 0 && src[pos] != '[' {
		pos--
	}
	if pos > 0 && src[pos-1] == ']' && src[pos] == '[' {
		end := strings.IndexByte(src[pos:], ']')
		if end > 0 && offset <= pos+end {
			return src[pos+1 : pos+end]
		}
	}
	return ""
}

func findInlineLinkAt(doc *Document, offset int) string {
	// Search for "(url)" after "]" around offset
	src := doc.Source
	pos := offset
	for pos > 0 && src[pos] != '(' {
		pos--
	}
	if pos > 0 && src[pos-1] == ']' && src[pos] == '(' {
		end := strings.IndexByte(src[pos:], ')')
		if end > 0 && offset <= pos+end {
			return src[pos+1 : pos+end]
		}
	}
	return ""
}
```

```go
package server

import (
	"github.com/tliron/glsp"
	protocol "github.com/tliron/glsp/protocol_3_16"
)

func handleDefinition(ctx *glsp.Context, params *protocol.DefinitionParams) (any, error) {
	doc := getDocument(params.TextDocument.URI)
	if doc == nil {
		return nil, nil
	}
	loc := computeDefinition(doc, params.Position)
	if loc == nil {
		return nil, nil
	}
	return *loc, nil
}

func computeDefinition(doc *Document, pos protocol.Position) *protocol.Location {
	offset := doc.PositionToOffset(pos)

	// Check if cursor is on a footnote reference
	if ref := findFootnoteRefAt(doc, offset); ref != "" {
		if def, ok := doc.FootnoteDefs[ref]; ok {
			return &protocol.Location{
				URI:   doc.URI,
				Range: def.Range,
			}
		}
	}

	// Check if cursor is on a link reference
	if ref := findLinkRefAt(doc, offset); ref != "" {
		if def, ok := doc.ReferenceDefs[ref]; ok {
			return &protocol.Location{
				URI:   doc.URI,
				Range: def.Range,
			}
		}
	}

	return nil
}
```
